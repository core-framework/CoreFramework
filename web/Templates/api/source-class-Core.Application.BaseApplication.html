200     /**
201      * Runs the application
202      *
203      * @throws \ErrorException
204      */
205     public function run()
206     {
207         /** @var \Core\CacheSystem\Cache $cache */
208         $this->cache = $cache = $this->get('Cache');
209 
210         if (isset($_GET['action']) && $_GET['action'] === 'clear_cache') {
211             $cache->clearCache();
212         }
213 
214         $path = $_SERVER['REQUEST_URI'];
215         $this->conf['$global']['routeKey'] = $routeKey = md5($path . '_route_' . session_id());
216         $this->conf['$global']['viewKey'] = $viewKey = md5($path . '_view_' . session_id());
217 
218         if ($cache->cacheExists($viewKey)) {
219             $this->status = self::STATUS_LOADING_FROM_CACHE;
220             $this->renderFromCache($viewKey);
221             return;
222         }
223 
224         $this->router = $cache->getCache($routeKey);
225 
226         if (!($this->router instanceof Router)) {
227             $this->router = $this->get('Router');
228             $this->requestedURI = $this->router->path;
229         }
230 
231         $this->parseRoute();
232 
233     }

249     /**
250      * Parses the URI route requested
251      *
252      * @throws \ErrorException
253      */
254     public function parseRoute()
255     {
256         $this->status = self::STATUS_HANDLING_REQUEST;
257         $this->routeParams = $routeParams = $this->router->resolve($this->global['useAestheticRouting']);
258 
259         if(!isset($routeParams['noCacheRoute']) || $routeParams['noCacheRoute'] === false) {
260             $this->cache->cacheContent($this->conf['$global']['routeKey'], $this->router, $this->ttl);
261         }
262 
263         $this->view = $this->get('View');
264         if ($this->router->customServe === true) {
265             $this->handleCustomServe($routeParams);
266         } else {
267             $this->loadController($routeParams);
268             if ($this->view->disabled !== true) {
269                 $this->view->render();
270                 $this->status = self::STATUS_SENT_RESPONSE;
271 
272                 if (( isset($this->conf['$global']['noCache']) && $this->conf['$global']['noCache'] === true) || (isset($routeParams['noCache']) && $routeParams['noCache']) === true) {
273                     return;
274                 }
275 
276                 $this->cache->cacheContent($this->conf['$global']['viewKey'], $this->view, $this->ttl);
277             }
278         }

282     /**
283      * Serve Custom content. This is to handle custom serving of view content
284      *
285      * @param array $routeParams
286      */
287     private function handleCustomServe($routeParams = [])
288     {
289         if (empty($routeParams)) {
290             return;
291         }
292 
293         $args = $routeParams['args'];
294         $routeVars = $routeParams['routeVars'];
295         $showHeader = isset($routeVars['showHeader']) && $routeVars['showHeader'] === true ? true : false;
296         $showFooter = isset($routeVars['showFooter']) && $routeVars['showFooter'] === true ? true : false;
297         $serveIframe = isset($routeVars['serveIframe']) && $routeVars['serveIframe'] === true ? true : false;
298         $fileName = $this->router->fileName;
299         $fileName = !empty($fileName) ? $fileName : 'index';
300         $fileExt = $this->router->fileExt;
301         $fileExt = !empty($fileExt) ? $fileExt : 'html';
302 
303         $referencePath = $this->router->referencePath;
304 
305         $rPathArr = explode('/', $referencePath);
306 
307         $realPath = $this->appPath . DS;
308 
309         foreach ($rPathArr as $part) {
310             $realPath .= $part . DS;
311         }
312 
313         if (!empty($args)) {
314             $key = key($args);
315             $realPath .= $args[$key];
316         } else {
317             $realPath .= $fileName . "." . $fileExt;
318         }
319 
320         if ($showHeader === true && $serveIframe === false && $fileExt === 'html') {
321             $this->view->showHeader = $showHeader;
322             $this->view->showFooter = $showFooter;
323             $this->view->tplEngine->assign('customServePath', $realPath);
324             $this->view->setDebugMode(false);
325             $this->loadController($routeVars);
326 
327         } elseif ($serveIframe === true && $showHeader === true) {
328             $this->view->showHeader = $showHeader;
329             $this->view->showFooter = $showFooter;
330             $this->view->setTemplateVars('iframeUrl', $referencePath);
331             $this->view->setDebugMode(false);
332             $this->loadController($routeVars);
333 
334         } else {
335             $this->setHeaders($fileExt);
336             include_once $realPath;
337             $this->view->disable();
338         }
339 
340     }

344      *
345      * @param array $routeParams
346      */
347     public function loadController($routeParams = [])
348     {
349         $this->status = self::STATUS_COMPUTING_RESPONSE;
350         if (empty($routeParams)) {
351             $routeParams['namespace'] = '\\Core\\Controllers';
352             $routeParams['controller'] = 'errorController';
353             $routeParams['method'] = 'pageNotFound';
354         }
355 
356         $class = $routeParams['namespace'] . "\\" . $routeParams['controller'];
357         $controllerObj = new $class($this->router, $this->view, $this->conf);
358         $controllerObj->$routeParams['method'](isset($routeParams['args']) ? $routeParams['args'] : null);
